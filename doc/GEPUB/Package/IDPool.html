def generate_key(param = {})
  prefix = param[:prefix] || ''
  suffix = param[:suffix] || ''
  count = [ param[:start] || 1, counter(prefix,suffix) || 1].max
  while (true)
    if param[:without_count]
      k = prefix + suffix
      count -= 1
      param.delete(:without_count)
    else
      k = prefix + count.to_s + suffix
    end
    if @pool[k].nil?
      set_counter(prefix,suffix, count + 1)
      return k
    end
    count += 1
  end

end
